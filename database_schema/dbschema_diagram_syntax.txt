enum input_type {
  text_label
  text_input
  number_input
  dropdown_input
}

Table admins {
  id integer [primary key]
  username varchar
  hashed_password varchar
}

Table users {
  id integer [primary key]
  username varchar
  hashed_password varchar
}

Table form_items {
  id integer [primary key]
  form_id integer
  input_type input_type
  form_item_value varchar
  form_item_response varchar
  form_item_user integer
}

Table forms {
  id integer [primary key]
  admin_id integer
  path varchar
}
